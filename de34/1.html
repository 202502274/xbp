          void setup() {
            Serial.begin(9600);
            pinMode(13,OUTPUT);
          }
          
          void loop() {
            int val = analogRead(1);
            Serial.println(val);

            // 暗いときLED点灯
            if (val > 700) {
              digitalWrite(13,HIGH);
            } else {
              digitalWrite(13,LOW);
            }

            delay(500);
          }
